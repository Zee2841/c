package com.example.c;

import android.os.Bundle;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText number;
    Button check;
    TextView result;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        number = findViewById(R.id.etNumber);
        check = findViewById(R.id.btnCheck);
        result = findViewById(R.id.tvResult);

        check.setOnClickListener(v -> {

            int num = Integer.parseInt(number.getText().toString());
            boolean isPrime = true;

            if (num <= 1) {
                isPrime = false;
            } else {
                for (int i = 2; i <= num / 2; i++) {
                    if (num % i == 0) {
                        isPrime = false;
                        break;
                    }
                }
            }

            if (isPrime) {
                result.setText(num + " is a Prime Number");
            } else {
                result.setText(num + " is Not a Prime Number");
            }
        });
    }
}

package com.example.c;

import android.content.Intent;
import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText name, middle, surname, phone, email;
    RadioGroup rg;
    Button save;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        name = findViewById(R.id.etName);
        middle = findViewById(R.id.etMiddle);
        surname = findViewById(R.id.etSurname);
        phone = findViewById(R.id.etPhone);
        email = findViewById(R.id.etEmail);
        rg = findViewById(R.id.radioGroup);
        save = findViewById(R.id.btnSave);

        save.setOnClickListener(v -> {

            String n = name.getText().toString();
            String m = middle.getText().toString();
            String s = surname.getText().toString();
            String p = phone.getText().toString();
            String e = email.getText().toString();

            if(n.isEmpty() || s.isEmpty() || p.isEmpty() || e.isEmpty() || rg.getCheckedRadioButtonId() == -1){
                Toast.makeText(MainActivity.this,
                        "Please fill all required fields",
                        Toast.LENGTH_SHORT).show();
            } else {

                RadioButton rb = findViewById(rg.getCheckedRadioButtonId());
                String course = rb.getText().toString();

                Intent i = new Intent(MainActivity.this, SecondActivity.class);

                i.putExtra("name", n);
                i.putExtra("middle", m);
                i.putExtra("surname", s);
                i.putExtra("phone", p);
                i.putExtra("email", e);
                i.putExtra("course", course);

                startActivity(i);
            }
        });
    }
}

package com.example.c;

import android.os.Bundle;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class SecondActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        ((TextView)findViewById(R.id.tvName))
                .setText("Name: " + getIntent().getStringExtra("name"));

        ((TextView)findViewById(R.id.tvMiddle))
                .setText("Middle Name: " + getIntent().getStringExtra("middle"));

        ((TextView)findViewById(R.id.tvSurname))
                .setText("Surname: " + getIntent().getStringExtra("surname"));

        ((TextView)findViewById(R.id.tvPhone))
                .setText("Phone: " + getIntent().getStringExtra("phone"));

        ((TextView)findViewById(R.id.tvEmail))
                .setText("Email: " + getIntent().getStringExtra("email"));

        ((TextView)findViewById(R.id.tvCourse))
                .setText("Course: " + getIntent().getStringExtra("course"));
    }
}
