package com.example.c;

import android.os.Bundle;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText etNumber;
    Button btnSum;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etNumber = findViewById(R.id.etNumber);
        btnSum = findViewById(R.id.btnSum);

        btnSum.setOnClickListener(v -> {

            String input = etNumber.getText().toString();

            if(input.isEmpty()){
                Toast.makeText(this,"Enter a number",Toast.LENGTH_SHORT).show();
                return;
            }

            int num = Integer.parseInt(input);
            int sum = 0;

            while(num > 0){
                sum += num % 10;
                num = num / 10;
            }

            Toast.makeText(this,"Sum of digits = " + sum,Toast.LENGTH_LONG).show();
        });
    }
}

main java
package com.example.c;

import android.app.DatePickerDialog;
import android.app.TimePickerDialog;
import android.content.Intent;
import android.os.Bundle;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;
import java.util.Calendar;

public class MainActivity extends AppCompatActivity {

    Button btnDate, btnTime, btnShow;
    String selectedDate = "";
    String selectedTime = "";

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        btnDate = findViewById(R.id.btnDate);
        btnTime = findViewById(R.id.btnTime);
        btnShow = findViewById(R.id.btnShow);

        Calendar calendar = Calendar.getInstance();

        btnDate.setOnClickListener(v -> {
            new DatePickerDialog(this, (view, y, m, d) -> {
                selectedDate = d + "/" + (m+1) + "/" + y;
            },
                    calendar.get(Calendar.YEAR),
                    calendar.get(Calendar.MONTH),
                    calendar.get(Calendar.DAY_OF_MONTH)).show();
        });

        btnTime.setOnClickListener(v -> {
            new TimePickerDialog(this, (view, h, m) -> {
                selectedTime = h + ":" + m;
            },
                    calendar.get(Calendar.HOUR_OF_DAY),
                    calendar.get(Calendar.MINUTE),
                    true).show();
        });

        btnShow.setOnClickListener(v -> {

            Intent i = new Intent(MainActivity.this, SecondActivity.class);
            i.putExtra("date", selectedDate);
            i.putExtra("time", selectedTime);
            startActivity(i);
        });
    }
}

package com.example.c;

import android.os.Bundle;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class SecondActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        TextView tv = findViewById(R.id.tvResult);

        String date = getIntent().getStringExtra("date");
        String time = getIntent().getStringExtra("time");

        tv.setText("Selected Date: " + date +
                "\nSelected Time: " + time);
    }
}
