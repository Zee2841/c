package com.example.c;

import android.os.Bundle;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText number;
    Button check;
    TextView result;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        number = findViewById(R.id.etNumber);
        check = findViewById(R.id.btnCheck);
        result = findViewById(R.id.tvResult);

        check.setOnClickListener(v -> {

            int num = Integer.parseInt(number.getText().toString());
            int sum = 0;

            for(int i = 1; i <= num/2; i++) {
                if(num % i == 0) {
                    sum = sum + i;
                }
            }

            if(sum == num && num != 0) {
                result.setText(num + " is a Perfect Number");
            } else {
                result.setText(num + " is Not a Perfect Number");
            }
        });
    }
}

package com.example.categoryspinner;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.ArrayAdapter;
import android.widget.AdapterView;
import android.widget.Spinner;

public class MainActivity extends AppCompatActivity {

    Spinner spinnerCategory, spinnerProduct;

    String[] categories = {"Select Category", "Electronics", "Clothing", "Food"};

    String[] electronicsProducts = {"Laptop", "Mobile", "Camera"};
    String[] clothingProducts = {"Shirt", "Jeans", "T-Shirt"};
    String[] foodProducts = {"Pizza", "Burger", "Pasta"};

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        spinnerCategory = findViewById(R.id.spinnerCategory);
        spinnerProduct = findViewById(R.id.spinnerProduct);

        // Category Adapter
        ArrayAdapter<String> categoryAdapter = new ArrayAdapter<>(
                this,
                android.R.layout.simple_spinner_item,
                categories);
        categoryAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
        spinnerCategory.setAdapter(categoryAdapter);

        // Category Selection Listener
        spinnerCategory.setOnItemSelectedListener(new AdapterView.OnItemSelectedListener() {
            @Override
            public void onItemSelected(AdapterView<?> parent, View view,
                                       int position, long id) {

                String[] selectedProducts = null;

                if (position == 1) {
                    selectedProducts = electronicsProducts;
                } else if (position == 2) {
                    selectedProducts = clothingProducts;
                } else if (position == 3) {
                    selectedProducts = foodProducts;
                }

                if (selectedProducts != null) {
                    ArrayAdapter<String> productAdapter = new ArrayAdapter<>(
                            MainActivity.this,
                            android.R.layout.simple_spinner_item,
                            selectedProducts);
                    productAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
                    spinnerProduct.setAdapter(productAdapter);
                }
            }

            @Override
            public void onNothingSelected(AdapterView<?> parent) {

            }
        });
    }
}
