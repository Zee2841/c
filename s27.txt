package com.example.gridviewdemo;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.ArrayAdapter;
import android.widget.GridView;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    GridView gridView;

    String[] items = {
            "One", "Two", "Three",
            "Four", "Five", "Six",
            "Seven", "Eight", "Nine"
    };

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        gridView = findViewById(R.id.gridView);

        ArrayAdapter<String> adapter = new ArrayAdapter<>(
                this,
                android.R.layout.simple_list_item_1,
                items
        );

        gridView.setAdapter(adapter);

        gridView.setOnItemClickListener((parent, view, position, id) -> {
            Toast.makeText(this,
                    "Clicked: " + items[position],
                    Toast.LENGTH_SHORT).show();
        });
    }
}

package com.example.foodorder;

import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.os.Bundle;
import android.widget.Button;
import android.widget.EditText;
import android.widget.RadioButton;
import android.widget.RadioGroup;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    EditText etName, etTable, etPhone;
    RadioGroup rgFoodType;
    RadioButton rbVeg, rbNonVeg;
    Button btnPlaceOrder;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etName = findViewById(R.id.etName);
        etTable = findViewById(R.id.etTable);
        etPhone = findViewById(R.id.etPhone);
        rgFoodType = findViewById(R.id.rgFoodType);
        rbVeg = findViewById(R.id.rbVeg);
        rbNonVeg = findViewById(R.id.rbNonVeg);
        btnPlaceOrder = findViewById(R.id.btnPlaceOrder);

        btnPlaceOrder.setOnClickListener(v -> {
            String name = etName.getText().toString().trim();
            String table = etTable.getText().toString().trim();
            String phone = etPhone.getText().toString().trim();
            int selectedFoodId = rgFoodType.getCheckedRadioButtonId();

            // Validation
            if (name.isEmpty() || table.isEmpty() || phone.isEmpty() || selectedFoodId == -1) {
                Toast.makeText(this, "Please fill all required fields", Toast.LENGTH_SHORT).show();
                return;
            }

            String foodType = selectedFoodId == R.id.rbVeg ? "Veg" : "Non-Veg";

            // Show AlertDialog
            new AlertDialog.Builder(this)
                    .setTitle("Order Received")
                    .setMessage("Your order has been received successfully!")
                    .setPositiveButton("OK", (dialog, which) -> {
                        // Open Second Activity
                        Intent intent = new Intent(MainActivity.this, OrderActivity.class);
                        intent.putExtra("name", name);
                        intent.putExtra("table", table);
                        intent.putExtra("phone", phone);
                        intent.putExtra("food", foodType);
                        startActivity(intent);
                    })
                    .show();
        });
    }
}

package com.example.foodorder;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.TextView;

public class OrderActivity extends AppCompatActivity {

    TextView tvOrderDetails;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_order);

        tvOrderDetails = findViewById(R.id.tvOrderDetails);

        String name = getIntent().getStringExtra("name");
        String table = getIntent().getStringExtra("table");
        String phone = getIntent().getStringExtra("phone");
        String food = getIntent().getStringExtra("food");

        String orderDetails = "Customer Name: " + name +
                "\nTable Number: " + table +
                "\nPhone Number: " + phone +
                "\nFood Type: " + food;

        tvOrderDetails.setText(orderDetails);
    }
}
